    **shared.tsx** -- Create defensive config extraction helpers that every primitive will import:

    ```typescript
    // Config extraction with type guards and fallbacks
    export function configString(config: Record<string, unknown>, key: string, fallback: string): string
    export function configNumber(config: Record<string, unknown>, key: string, fallback: number): number
    export function configBool(config: Record<string, unknown>, key: string, fallback: boolean): boolean
    export function configArray<T>(config: Record<string, unknown>, key: string): T[]
    ```

    - `configString`: returns `string` if `typeof val === 'string'`, else `fallback`
    - `configNumber`: returns `number` if `typeof val === 'number' && Number.isFinite(val)`, else `fallback`
    - `configBool`: returns `boolean` if `typeof val === 'boolean'`, else `fallback`
    - `configArray`: returns `T[]` if `Array.isArray(val)`, else empty array `[]`

    **Color resolution map** for chart primitives:
    ```typescript
    export const COLOR_MAP: Record<string, string> = {
      accent: 'var(--accent)',
      'accent-dim': 'var(--accent-dim)',
      secondary: 'var(--secondary)',
      'secondary-dim': 'var(--secondary-dim)',
      tertiary: 'var(--tertiary)',
      'tertiary-dim': 'var(--tertiary-dim)',
      red: 'var(--red)',
      amber: 'var(--amber)',
      green: 'var(--secondary)',
      blue: 'var(--tertiary)',
    }

    export function resolveColor(key: string): string {
      return COLOR_MAP[key] ?? 'var(--accent)'
    }
    ```
