<objective>
Create the Widget Picker panel (add new widgets by category with search and size presets), RecycleBin drawer (recover removed widgets), and WidgetConfigPanel (per-widget settings via gear icon). These are the three widget management surfaces beyond the grid itself.

Purpose: Without the widget picker, users cannot add new widgets after initial setup. Without the recycle bin, widget removal is destructive. Without config panels, widgets have no per-instance customization. Together these complete the edit mode interaction set.

Output: `WidgetPicker.tsx`, `WidgetPickerCard.tsx`, `RecycleBin.tsx`, `WidgetConfigPanel.tsx`, tests.
</objective>

export interface WidgetDefinition {
  id: string

  description: string
  icon: string

}

export interface WidgetConfigSchema {
  fields: Array<{
    key: string; label: string;
    type: 'text' | 'number' | 'toggle' | 'select' | 'slider';
    default: unknown;
    options?: Array<{ label: string; value: unknown }>;
    min?: number; max?: number;
  }>
}

export interface WidgetBundle {
  id: string; name: string; description: string; widgetIds: string[];
}

export function getWidget(id: string): WidgetDefinition | undefined
export function getWidgetsByCategory(): Record<string, WidgetDefinition[]>
