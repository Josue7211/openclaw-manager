     ```typescript
     export function setBlendPosition(position: number | undefined) {
       if (position === undefined) {
         mutate(s => {
           const next = { ...s }
           delete next.blendPosition
           return next
         })
       } else {
         const clamped = Math.max(0, Math.min(1, position))
         mutate(s => ({ ...s, blendPosition: clamped }))
       }
       applyThemeFromState()
     }
     ```
