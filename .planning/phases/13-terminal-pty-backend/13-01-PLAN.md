```rust
static WS_CONNECTIONS: AtomicUsize = AtomicUsize::new(0);
const MAX_WS_CONNECTIONS: usize = 5;

struct WsConnectionGuard;
impl WsConnectionGuard {
    fn try_new() -> Option<Self> {
        loop {
            let current = WS_CONNECTIONS.load(Ordering::Acquire);
            if current >= MAX_WS_CONNECTIONS { return None; }
            if WS_CONNECTIONS.compare_exchange(current, current + 1, Ordering::AcqRel, Ordering::Acquire).is_ok() {
                return Some(Self);
            }
        }
    }
}
impl Drop for WsConnectionGuard {
    fn drop(&mut self) { WS_CONNECTIONS.fetch_sub(1, Ordering::AcqRel); }
}
```

/// Build the sanitized environment for PTY sessions.
/// Returns only whitelisted vars + TERM=xterm-256color.
/// This function is separate from build_pty_command() so unit tests can verify it.
fn build_sanitized_env() -> Vec<(String, String)> {
    let mut env = Vec::new();
    for key in PTY_SAFE_ENV_VARS {
        if let Ok(val) = std::env::var(key) {
            env.push((key.to_string(), val));
        }
    }
    env.push(("TERM".to_string(), "xterm-256color".to_string()));
    env
}

fn build_pty_command() -> CommandBuilder {
    let shell = detect_shell();
    let mut cmd = CommandBuilder::new(&shell);
    // CRITICAL: Clear ALL env vars first -- Tauri process has 30+ secrets loaded by dotenvy
    cmd.env_clear();
    // Whitelist only safe system variables
    for (key, val) in build_sanitized_env() {
        cmd.env(key, val);
    }
    cmd
}

**Section G: PtyCleanup struct with process group kill on Drop**

This is the CRITICAL piece. Must go BEFORE the WebSocket handler since it uses it.

```rust
struct PtyCleanup {
    child: Box<dyn portable_pty::Child + Send>,
    #[cfg(unix)]
    pgid: Option<i32>,
}

impl Drop for PtyCleanup {
    fn drop(&mut self) {
        // Step 1: Kill the child process directly
        let _ = self.child.kill();

        // Step 2: Kill entire process group (catches subprocesses like vim, htop, etc.)
        #[cfg(unix)]
        if let Some(pgid) = self.pgid {
            if pgid > 0 {
                // kill with negative pid sends signal to the entire process group
                unsafe { libc::kill(-(pgid as i32), libc::SIGKILL); }
            }
        }

        // Step 3: Reap the zombie process to prevent PID table leak
        let _ = self.child.wait();

        info!("PTY session cleaned up (process group killed)");
    }
}
```

**Section I: Bidirectional WebSocket-PTY relay (`handle_terminal_ws`)**

This is the core function. Key architecture decision: use dedicated OS threads (via `std::thread::spawn`) for PTY I/O because `portable-pty` returns `Box<dyn Read + Send>` and `Box<dyn Write + Send>` which are blocking. Bridging to async via `tokio::sync::mpsc` channels.

```rust
async fn handle_terminal_ws(socket: WebSocket, _guard: PtyConnectionGuard) {
    // 1. Spawn PTY
    let pty_system = native_pty_system();
    let pair = match pty_system.openpty(PtySize {
        rows: 24,
        cols: 80,
        pixel_width: 0,
        pixel_height: 0,
    }) {
        Ok(p) => p,
        Err(e) => {
            error!("Failed to open PTY: {e}");
            return;
        }
    };

    // 2. Build sanitized command and spawn
    let cmd = build_pty_command();
    let child = match pair.slave.spawn_command(cmd) {
        Ok(c) => c,
        Err(e) => {
            error!("Failed to spawn shell: {e}");
            return;
        }
    };

    // 3. CRITICAL: drop slave immediately to prevent FD leak and EOF hang
    drop(pair.slave);

    // 4. Get process group leader for cleanup
    #[cfg(unix)]
    let pgid = pair.master.process_group_leader();
    #[cfg(not(unix))]
    let pgid: Option<i32> = None;

    // 5. Get reader and writer from master
    let reader = match pair.master.try_clone_reader() {
        Ok(r) => r,
        Err(e) => {
            error!("Failed to clone PTY reader: {e}");
            return;
        }
    };
    let writer = match pair.master.take_writer() {
        Ok(w) => w,
        Err(e) => {
            error!("Failed to take PTY writer: {e}");
            return;
        }
    };

    // 6. Store cleanup handle -- master must stay alive for resize
    let master = pair.master;
    let cleanup = PtyCleanup {
        child,
        #[cfg(unix)]
        pgid,
    };

    // 7. Split WebSocket
    let (ws_sender, mut ws_receiver) = socket.split();
    let ws_sender = std::sync::Arc::new(tokio::sync::Mutex::new(ws_sender));

    // 8. PTY reader OS thread -> mpsc channel -> async WS sender
    let (pty_read_tx, mut pty_read_rx) =
        tokio::sync::mpsc::channel::<Vec<u8>>(32);

    std::thread::spawn(move || {
        let mut reader = reader;
        let mut buf = vec![0u8; 4096];
        loop {
            match reader.read(&mut buf) {
                Ok(0) => break, // EOF -- shell exited
                Ok(n) => {
                    if pty_read_tx.blocking_send(buf[..n].to_vec()).is_err() {
                        break; // receiver dropped (WS closed)
                    }
                }
                Err(_) => break,
            }
        }
    });

    // 9. PTY writer channel -- async side sends, OS thread writes
    let (pty_write_tx, pty_write_rx) =
        std::sync::mpsc::channel::<Vec<u8>>();

    std::thread::spawn(move || {
        let mut writer = writer;
        for data in pty_write_rx {
            if writer.write_all(&data).is_err() {
                break;
            }
        }
    });

    // 10. Async task: forward PTY output to WebSocket as Binary
    let ws_sender_clone = ws_sender.clone();
    let read_task = tokio::spawn(async move {
        while let Some(data) = pty_read_rx.recv().await {
            let mut sender = ws_sender_clone.lock().await;
            if sender.send(Message::Binary(data)).await.is_err() {
                break; // WS closed
            }
        }
    });

    // 11. Async task: forward WebSocket input to PTY + handle resize
    let write_task = tokio::spawn(async move {
        while let Some(Ok(msg)) = ws_receiver.next().await {
            match msg {
                Message::Text(text) => {
                    // Try to parse as JSON control command
                    if let Ok(cmd) = serde_json::from_str::<TerminalCommand>(&text) {
                        match cmd {
                            TerminalCommand::Resize { cols, rows } => {
                                let _ = master.resize(PtySize {
                                    rows,
                                    cols,
                                    pixel_width: 0,
                                    pixel_height: 0,
                                });
                            }
                            TerminalCommand::Input { data } => {
                                let _ = pty_write_tx.send(data.into_bytes());
                            }
                        }
                    } else {
                        // Not JSON -- treat as raw text input
                        let _ = pty_write_tx.send(text.into_bytes());
                    }
                }
                Message::Binary(data) => {
                    let _ = pty_write_tx.send(data);
                }
                Message::Close(_) => break,
                _ => {}
            }
        }
        // Drop pty_write_tx to signal the writer thread to exit
    });

    // 12. Wait for either task to finish, then clean up
    tokio::select! {
        _ = read_task => { info!("PTY read task ended"); }
        _ = write_task => { info!("PTY write task ended"); }
    }

    // cleanup is dropped here, which kills process group + reaps zombie
    // _guard is dropped here, which decrements PTY_CONNECTIONS counter
    drop(cleanup);
}
```

NOTE on `master` ownership: The `master` reference is moved into the `write_task` closure for resize support. If the compiler complains about `master` being both in `PtyCleanup` and moved into the task, restructure: do NOT put master into PtyCleanup. Instead, keep PtyCleanup holding only the child and pgid. The master will be dropped when `write_task` ends (which happens before the `drop(cleanup)` call since `tokio::select!` cancels the other future). This is safe because `kill(-pgid, SIGKILL)` does not need the master FD.
