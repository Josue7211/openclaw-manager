```typescript
const addMutation = useMutation({
  mutationFn: async (text: string) => { await api.post('/api/todos', { text }) },
  onMutate: async (text) => {
    await queryClient.cancelQueries({ queryKey: queryKeys.todos })
    const prev = queryClient.getQueryData(queryKeys.todos)
    queryClient.setQueryData(queryKeys.todos, (old) => ({
      ...old, todos: [...(old?.todos || []), { id: 'temp-' + Date.now(), text, done: false }],
    }))
    return { prev }
  },
  onError: (_err, _text, ctx) => { if (ctx?.prev) queryClient.setQueryData(queryKeys.todos, ctx.prev) },
  onSettled: () => invalidateTodos(),
})
```

**A. Create `frontend/src/hooks/useAgents.ts`:**

Extract a dedicated hook following the useTodos.ts pattern exactly. The hook provides:

```typescript
import { useCallback } from 'react'
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import { api } from '@/lib/api'
import { queryKeys } from '@/lib/query-keys'
import type { Agent, CreateAgentPayload, AgentAction } from '@/pages/agents/types'
```

1. `useAgents()` function that returns:
   - `agents: Agent[]` (from useQuery GET /api/agents, returns `data?.agents ?? []`)
   - `loading: boolean` (isLoading from useQuery)
   - `createMutation` -- optimistic: adds a temp agent with `id: 'temp-' + Date.now()`, `status: 'idle'`, `sort_order: 999`, `name: display_name.toLowerCase()`, `created_at/updated_at: new Date().toISOString()`. Calls `api.post<{ agent: Agent }>('/api/agents', payload)`. On error: rollback. On settled: invalidate.
   - `updateMutation` -- optimistic: updates matching agent in cache. Calls `api.patch<{ agent: Agent }>('/api/agents', { id, ...fields })`. On error: rollback. On settled: invalidate.
   - `deleteMutation` -- optimistic: removes agent from cache. Calls `api.del('/api/agents', { id })`. On error: rollback. On settled: invalidate.
   - `actionMutation` -- NOT optimistic (fire-and-forget to gateway). Calls `api.post('/api/agents/action', { id, action })`. On settled: invalidate agents.
   - `invalidateAgents` -- `() => queryClient.invalidateQueries({ queryKey: queryKeys.agents })`

2. The useQuery should be disabled when `isDemoMode()` is true (import from `@/lib/demo-data`).
